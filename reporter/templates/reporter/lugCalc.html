# Lifting Lug {{ lugParams.component_id }}

Lift Forces
Fr = [W / cos(Φ1)]*(1 - x1 / (x1 + x2))
= [{{ lugParams.W }} / cos({{ lugParams.phi }})]*(1 - {{ lugParams.x_1 }} / ({{ lugParams.x_1 }} + {{ lugParams.x_2 }}))
= {{ lugParams.F_r }}lbf

Lug Pin Diameter - Shear stress
dreqd = (2*Fr / (Π*σ))0.5
= (2*{{ lugParams.F_r }} / (Π*{{ lugParams.sigma_s }}))0.5
= {{ lugParams.d_reqd }}in
dreqd / Dp = {{ lugParams.d_reqd }} / {{ lugParams.D_p }} = {{ lugParams.diameter_ratio }}
σ = Fr / A
= Fr / (2*(0.25*Π*Dp2))
= {{ lugParams.F_r }} / (2*(0.25*Π*{{ lugParams.D_p }}2))
= {{ lugParams.sigma_sd_calc }}psi
σ / σs = {{ lugParams.sigma_sd_calc }} / {{ lugParams.sigma_s }}

Lug Thickness - Tensile stress
treqd = Fr / ((L - d)*σt)
= {{ lugParams.F_r }} / (({{ lugParams.L }} - {{ lugParams.d }})*{{ lugParams.sigma_t }})
= {{ lugParams.t_reqd_tensile }}
treqd / t = {{ lugParams.t_reqd_tensile }} / {{ lugParams.t }} = {{ lugParams.thickness_ratio }}
σ = Fr / A
= Fr / ((L-d)*t)
= {{ lugParams.F_r }} / (({{ lugParams.L }} - {{ lugParams.d }})*{{ lugParams.t }})
= {{ lugParams.sigma_t_calc }}
σ / σt = {{ lugParams.sigma_t_calc }} / {{ lugParams.sigma_t }} = {{ lugParams.sigma_t_ratio }}

Lug Thickness - Bearing stress
treqd = Fv / (Dp*sigmap)
= {{ lugParams.F_r }} / ({{ lugParams.D_p }}*{{ lugParams.sigma_p }})
= {{ lugParams.t_reqd_bearing }}
treqd / t = {{ lugParams.t_reqd_bearing }} / {{ lugParams.t }} = {{ lugParams.thickness_ratio }}
σ = Fv / Abearing
= Fv / (Dp*(t))
= {{ lugParams.F_r }} / ({{ lugParams.D_p }}*{{ lugParams.t }})
= {{ lugParams.sigma_b_calc }}
σ / σp = {{ lugParams.sigma_b_calc }} / {{ lugParams.sigma_b }} = {{ lugParams.sigma_b_ratio }}

Lug Thickness - Shear Stress
treqd = [Fv / σs] / (2*Lshear)
= [{{ lugParams.F_r }} / {{ lugParams.sigma_s }}] / (2*{{ lugParams.L_shear }})
= {{ lugParams.t_reqd_shear }}
treqd / t = {{ lugParams.t_reqd_shear }} / {{ lugParams.t }} = {{ lugParams.thickness_ratio }}

= Fv / (2*t*Lshear)
= {{ lugParams.F_r }} / (2*{{ lugParams.t }}*{{ lugParams.L_shear }})
= {{ lugParams.sigma_s_calc }}
σ / σp = {{ lugParams.sigma_s_calc }} / {{ lugParams.sigma_s }} = {{ lugParams.sigma_s_ratio }}

Shear Stress length
φ = 55*Dp / d
= 55*{{ lugParams.D_p }} / {{ lugParams.d }}
= {{ lugParams.phi_shear }}
Lshear = (H - a2 - 0.5*d) + 0.5*Dp*(1 - cos(φ))
= ({{ lugParams.H }} - {{ lugParams.a2 }} - 0.5*{{ lugParams.d }}) + 0.5*{{ lugParams.D_p }}*(1 - cos({{ lugParams.phi_shear }}))
